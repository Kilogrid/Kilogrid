291 ISR(ANALOG_COMP_vect) {
292 
293  cli();
294 
295  uint16_t timer = TCNT1; // save Timer 1 state
296 
297  rx_busy = 1; // inhibits parts in the other interrupts
298 
299  if(rx_leadingbit) { // start bit received.
300  rx_timer_on(); // switch on timeout timer
301  rx_bytevalue = 0; // start at byte 0
302  rx_leadingbit = 0; // reset rx_leadingbit flag
303  if (rx_leadingbyte) {
304  ADC_FINISH_CONVERSION();
305  rx_dist.high_gain = ADCW; // measure the high gain value of the IR signal at the first bit of the first byte received
306 
307  // change to low gain signal until next bit arrives
308  SELECT_IR_RECEIVER(current_rx_cell, LGAIN);
309 
310  }
311  }
312  else { // non starting bit received
313 
314  if (timer <= rx_bitcycles/2 || timer >= rx_bitcycles*9+rx_bitcycles/2) {
315  rx_timer_off();
316  rx_leadingbit = 1; // prepare for next start bit to arrive
317  rx_leadingbyte = 1;
318  rx_busy = 0;
319  SELECT_IR_RECEIVER(current_rx_cell, HGAIN);
320 
321  }
322  else {
323  // NOTE: The following code avoids a division which takes
324  // too many clock cycles and throws off the interrupt.
325  const uint16_t M = ((1L<<16)+rx_bitcycles-1)/rx_bitcycles;
326  uint8_t bitindex = ((uint32_t)(timer-rx_bitcycles/2)*M)>>16;
327  if (bitindex <= 7) {
328  rx_bytevalue |= (1<<bitindex);
329  }
330  else{ // stop bit of a byte received
331  rx_leadingbit = 1; // prepare for next start bit to arrive
332  if (rx_leadingbyte) {
333  ADC_FINISH_CONVERSION();
334  rx_dist.low_gain = ADCW; // measure the low value of the gain at the first byte of a message
335 
336  // Switch back to high gain
337  SELECT_IR_RECEIVER(current_rx_cell, HGAIN);
338 
339  if (rx_bytevalue != 0) { // Collision detected.
340  rx_timer_off();
341  rx_leadingbyte = 1;
342  rx_busy = 0;
343 
344  //TCCR2B |= (1 << CS20) | (1 << CS21) | (1 << CS22); // restart mux timer
345  }

349  }
