15 /*
16  * These macros allow the user to set ports directions and bits based on the generic names
17  * of the signals on the cell. Generic names are: IRRXCOM_A,D ; IRRXADDR0,1,2 ; IRTXCOM ; IRTXADDR0.1 ; etc...
18  *
19  * Usage: SET_AS_OUTPUT(DEBUG_1);
20  *
21  */
22 
23 /*
24  * @brief Write a 1 in place of the specified bit in the direction register.
25  */
26 #define SET_AS_OUTPUT(o)\
27  o##dir |= (1 << o##bit);
28 
29 /*
30  * @brief Write a 0 in place of the specified bit in the direction register
31  */
32 #define SET_AS_INPUT(o)\
33  o##dir &= ~(1 << o##bit);
34 
35 /*
36  * @brief Write a 1 in place of the specified bit in the port register
37  */
38 #define PIN_HIGH(o)\
39  o##port |= (1 << o##bit);
40 
41 /*
42  * @brief Write a 0 in place of the specified bit in the port register
43  */
44 #define PIN_LOW(o)\
45  o##port &= ~(1 << o##bit);
46 
47 /*
48  * @brief Check if the specified bit is set.
49  */
50 #define IS_SET(o)\
51 (o##port & (1 << o##bit)) != 0
