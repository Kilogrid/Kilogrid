9  * Simple AVR demonstration. Controls a LED that can be directly
10  * connected from OC1/OC1A to GND. The brightness of the LED is
11  * controlled with the PWM. After each period of the PWM, the PWM
12  * value is either incremented or decremented, that's all.
13  *
14  * $Id: demo.c,v 1.9 2006/01/05 21:30:10 joerg_wunsch Exp $
15  */
16 
17 #include <inttypes.h>
18 #include <avr/io.h>
19 #include <avr/interrupt.h>
20 #include <avr/sleep.h>
21 
22 #include "i2c_master.h"
23 
24 const uint8_t PCA9635_i2c_addr = 0x15; // TODO: to be changed to struct later
25 
26 void init_PCA9635(void) /* Note [6] */
27 {
28  I2C_start((PCA9635_i2c_addr << 1) + I2C_WRITE);
29 
30  //Autoincrement ALL registers, start at reg 0 - we're initting
31  I2C_write(0x80);
32 
33  //Reg 0x00 = MODE1 - set to 0x80 - autoincrement enabled (this is readonly?), do not respond to subaddresses or allcall
34  I2C_write(0x80);
35 
36  //Reg 0x01 = MODE2 - set to 0x02 - noninverted logic state, open-drain
37  I2C_write(0x02);
38 
39  //Reg 0x02-0x11 - PWM0-15 - LED brightnesses, start at low brightness (0x05)
40 
41  I2C_write(0x05);
42  I2C_write(0x05);
43  I2C_write(0x05);
44  I2C_write(0x05);
45  I2C_write(0x05);
46  I2C_write(0x05);
47  I2C_write(0x05);
48  I2C_write(0x05);
49  I2C_write(0x05);
50  I2C_write(0x05);
51  I2C_write(0x05);
52  I2C_write(0x05);
53  I2C_write(0x05);
54  I2C_write(0x05);
55  I2C_write(0x05);
56  I2C_write(0x05);
57 
58  //Reg 0x12 - Group PWM - should not matter
59  I2C_write(0xff);
60 
61  //Reg 0x13 - Group Freq - should not matter
62  I2C_write(0x00);
63 
64  //Reg 0x14-0x17 - LED Output State Control - all 0xAA (output state for each LED = 0x2)
65  I2C_write(0xaa);
66  I2C_write(0xaa);
67  I2C_write(0xaa);
68  I2C_write(0xaa);
69 
70  //Reg 0x18-0x1b - Subaddressing stuff, doesn't matter, just stop the xfer
71  I2C_stop();
72 
73 }
