340 static inline void process_CAN_message() {
341  uint8_t i, j = 0;
342 
343  CAN_msg_type = CAN_message_rx_buffer.data[0];
344  uint8_t current_CAN_msg_number = CAN_message_rx_buffer.data[1];
345 
346  // transfer the content of the CAN msg buffer into the CAN_message struct to avoid race conditions
347  if(CAN_msg_type == CAN_KILO_BOOTPAGE){
348  CAN_message_bootpage = CAN_message_rx_buffer;
349  }
350  else{
351  CAN_message_rx = CAN_message_rx_buffer;
352  }
353 
354  switch(CAN_msg_type){
355 
356  case CAN_FORWARD_IR_MESSAGE_STOP:
357 
358  init_IR_message(&IR_setup_message);
359 
360  disable_IR_tx(CELL_00);
361  disable_IR_tx(CELL_01);
362  disable_IR_tx(CELL_02);
363  disable_IR_tx(CELL_03);
364 
365  module_state = MODULE_IDLE;
366 
367  break;
368 
369  case CAN_KILO_BOOTPAGE_SIZE:
370  page_total = CAN_message_rx.data[2]; // store the size of the current Kilobot program
371  break;
372 
373  case CAN_KILO_BOOTPAGE:
374  // bootpage is faulty if the sequence number is not contiguous
375  if(current_CAN_msg_number != 0
376  && current_CAN_msg_number != (previous_CAN_msg_number+1))
377  {
378  bootpage_faulty = 1;
379  }
380 
381  // if bootpage is faulty then drop the rest of the bootpage
382  if(bootpage_faulty) {
383  break;
384  }
385 
386  // current_CAN_msg_number is the bootpage data offset, in bytes
387  page_offset = (current_CAN_msg_number * 6);
388 
389  if(current_CAN_msg_number < (CAN_MESSAGES_PER_BOOTPAGE-1)){
390  for(j = 0; j < 6; j++){
391  current_bootpage.data[page_offset + j] = CAN_message_bootpage.data[2 + j];
392  }
393  }
394  else if(current_CAN_msg_number == (CAN_MESSAGES_PER_BOOTPAGE-1)){ // if the message is the last CAN message of current bootpage: only copy the two remaining words (128 - 126 = 2 bytes)
395  for(j = 0; j < 2; j++){
396  current_bootpage.data[page_offset + j] = CAN_message_bootpage.data[2 + j];
397  }
398  }
399 
400  if(current_CAN_msg_number == (CAN_MESSAGES_PER_BOOTPAGE-1)){ // bootpage reception complete
401  page_count++;
402 
403  module_state = MODULE_BOOTLOADING_KILOBOTS; // switch to MODULE_BOOTLOADING_KILOBOTS state to send the bootpage in the main loop
404 
405  bootpage_sent = 0;
406 
407  } // if bootpage received
408 
409  previous_CAN_msg_number = current_CAN_msg_number;
410 
411  break;
412 
413  case CAN_KILO_BOOTPAGE_NUMBER:
414  bootpage_faulty = 0;
415  previous_CAN_msg_number = 0;
416  page_address = CAN_message_rx.data[2]; // store address of current bootpage
417  module_state = MODULE_WAITING_FOR_KILOBOT_BOOTPAGE;
418  break;
419 
420  case CAN_FORWARD_IR_MESSAGE: // IR message to be forwarded: receive it and broadcast to every cell
421  if(current_CAN_msg_number == 0){ // first part of the IR message
422 
423  // copy CAN message into IR message - TODO change this using mem copy
424  // i points to the CAN message data byte
425  // this step copies 6 bytes of overall 9 into the setup message data buffer
426  for (i = 0; i < 6; i++) {
427  IR_setup_message.data[i] = CAN_message_rx.data[i+2];
428  }
429 
430  }
431  else if(current_CAN_msg_number == 1){ // second part of the IR message
432  // there are 6 bytes in the IR_setup_message.data buffer
433  // remaining IR data bytes to copy from CAN message: 9 - (CAN_MESSAGE_SIZE - 2) = 3
434  // start from byte 2 in the CAN message for 3 bytes and copy to the IR_setup_message.data buffer from byte 6
435  for (i = 0; i < 3; i++){
436  IR_setup_message.data[6+i] = CAN_message_rx.data[i+2];
437  }
438  // i is incremented and points to the last byte of the IR message, which is the type
439  IR_setup_message.type = CAN_message_rx.data[i+2];
440 
441  // if this is a calibration message but we are not part of the calibration area, skip processing
442  if(IR_setup_message.type == CALIB && module_calib_restricted == 1) {
443  if( module_uid_y_coord < module_calib_y_min ||
444  module_uid_y_coord > module_calib_y_max ||
445  module_uid_x_coord < module_calib_x_min ||
446  module_uid_x_coord > module_calib_x_max) {
447  return;
448  }
449  }
450 
451  module_state = MODULE_SETUP_KILOBOTS; // will enter the MODULE_SETUP_KILOBOTS state after this ISR (ie. after receiving the entire IR message)
452 
453  // display color
454  switch(IR_setup_message.type){
455  case RESET:
456  c = GREEN;
457  break;
458  case BOOT:
459  c = BLUE;
460  bootloading_current_cell = CELL_00; // start bootloading on cell CELL_00
461  page_count = 0; // reset page count
462  break;
463  case BOOTPGM_PAGE:
464  c = CYAN;
465  break;
466  case SLEEP:
467  c = WHITE;
468  break;
469  case RUN:
470  c = MAGENTA;
471  break;
472  case VOLTAGE:
473  c = YELLOW;
474  break;
475 
476  // NOTE (ATFER REV 812): THE MODULE SHOULD NOT RECEIVE ANY CAN_FORWARD_IR_MESSAGE CONTAINING A BOOTPGM_SIZE IR MESSAGE TYPE (SENT AS A SEPARATE CAN MESSAGE AT THE BEGINNING OF THE BOOTLOADING PROCEDURE)
477  case BOOTPGM_SIZE:
478  c = CYAN;
479  break;
480 
481  case CHARGE:
482  c = RED;
483  break;
484 
485  case CALIB:
486  case GO_STRAIGHT:
487  case TURN_LEFT:
488  case TURN_RIGHT:
489  c = YELLOW;
490  break;
491 
492  default:
493  c = LED_OFF;
494  break;
495  }
496 
497  }
498 
499  break;
500 
501  case CAN_MODULE_BOOT:
502  goto_bootloader();
503  break;
504 
505  case CAN_MODULE_IDLE:
506  module_state = MODULE_IDLE;
507  break;
508 
509  case CAN_MODULE_SETUP:
510  module_state = MODULE_SETUP;
511  break;
512 
513  case CAN_MODULE_RUN:
514  module_state = MODULE_RUNNING;
515  break;
516 
517  case CAN_MODULE_CONFIG_SIZE:
518  // size of the configuration data in bytes
519  configuration_size = CAN_message_rx.data[2];
520 
521  // number of CAN messages
522  configuration_total = CAN_message_rx.data[2] / 6;
523  if(CAN_message_rx.data[2] % 6 != 0) {
524  configuration_total += 1;
525  }
526 
527  configuration_faulty = 0;
528  configuration_msg_counter = 0;
529  module_state = MODULE_WAITING_FOR_CONFIGURATION_DATA;
530  c = RED;
531 
532  break;
533 
534  case CAN_MODULE_CONFIG_TRANSFER:
535  if(module_state != MODULE_WAITING_FOR_CONFIGURATION_DATA ||
536  configuration_faulty){
537  return;
538  }
539 
540  if(current_CAN_msg_number != configuration_msg_counter) {
541  configuration_faulty = 1;
542  return;
543  }
544 
545  for(i = 0; i < 6; i++) {
546  //multiple messages can address a module, data is collated
547  configuration[configuration_msg_counter*6 + i] = CAN_message_rx.data[i+2];
548  }
549 
550  configuration_msg_counter += 1;
551 
552  if(current_CAN_msg_number == configuration_total-1) {
553  received_setup = 0;
554  c = GREEN;
555  module_state = MODULE_SETUP;
556  }
557  break;
558 
559  case CAN_TRACKING_REQ:
560  sending_tracking_data = 1;
561  sent_tracking_header = 0;
562  messages_to_send = RB_size(CAN_message_tx_buffer);
563  break;
564 
565  default:
566  module_CAN_message_rx(&CAN_message_rx); // transfer CAN message to the user. The user can receive all other CAN messages.
567  break;
568  }
569 }

576

579

581

593

598  module_calib_y_max = eeprom_read_byte((uint8_t *)MODULE_EEPROM_CALIB_Y_MAX);
599

601  CAN_address_to_dispatcher.x = module_uid_x_coord;
602  CAN_address_to_dispatcher.y = module_uid_y_coord;
603  CAN_address_to_dispatcher.type = ADDR_DISPATCHER;

615

618  init_module_LED();
619  init_module_CAN(module_uid_x_coord, module_uid_y_coord);
620  init_module_IR();
621 
622  brightness_dir = 1; // increasing
623  configuration_size = 0;
624 
625  #ifdef ITF
626  // do not display animation. This will be part of the test routines in the main loop
627 
628  _delay_ms(10000); // waiting time before the test begins
629 
630  module_state = MODULE_RUNNING; // directly go to the main!
631 
632  #else
633  // display animation
634  for(uint8_t j = 0; j < 3; j++){ // repeat 3 times
635 
636  // light up all LEDs in sequence
637  for(uint8_t i = 0; i < 4; i++){
638  set_LED(i, 1 << j);
639  _delay_ms(50);
640  }
641  // switch off all LEDs in sequence
642  for(uint8_t i = 0; i < 4; i++){
643  set_LED(i, LED_OFF);
644  _delay_ms(50);
645  }
646  }
647  #endif
648 
649  uint8_t i;
650  for(i=0; i<MODULE_CONFIGURATION_BUFFER_SIZE; i++) {
651  configuration[i] = 0;
652  }
653 
654  cprints("Module initialized.");
655 
656  sei(); // exit critical section - enable interrupts

659 void module_start(void (*setup)(void), void (*loop)(void)) {
660  uint8_t i, j;
661 
662  #ifdef ITF // force setup() once before executing main loop
663  setup();
664  #endif
665 
666  /**** MODULE MAIN LOOP AND INTERNAL STATE MACHINE ****/
667  while(1){
668 
669  //#define ULTRA_VERBOSE_ON
670  #ifdef ULTRA_VERBOSE_ON
671  cprinti(module_state);
672  #endif
673 
674  switch(module_state){
675  case MODULE_INIT:
676  module_state = MODULE_IDLE; // initial state after module initialization
677 
678  if(module_autostart) {
679  module_state = MODULE_SETUP;
680  }
681  break;
682 
683  case MODULE_IDLE:
684  _delay_ms(70);
685 
686  if(brightness_idle == 20){
687  brightness_dir = -1;
688  }
689  else if(brightness_idle == 0){
690  brightness_dir = 1;
691  }
692 
693  brightness_idle += brightness_dir;
694 
695  set_all_LEDs_with_brightness(WHITE, brightness_idle);
696 
697  break; // wait for commands to arrive
698 
699  case MODULE_SETUP_KILOBOTS:
700 
701  send_IR_message(&IR_setup_message, CELL_00);
702  send_IR_message(&IR_setup_message, CELL_01);
703  send_IR_message(&IR_setup_message, CELL_02);
704  send_IR_message(&IR_setup_message, CELL_03);
705 
706  set_all_LEDs(c); // display state
707 
708  break;
709 
710  case MODULE_WAITING_FOR_KILOBOT_BOOTPAGE:
711  case MODULE_WAITING_FOR_CONFIGURATION_DATA:
712  break;
713 
714  case MODULE_BOOTLOADING_KILOBOTS:
715 
716  if(bootpage_sent == 0) {
717 
718  cli(); // enter critical section
719 
720  // tests
721  #ifdef VERBOSE_ON
722  // test: page_address == 0 at some point here?
723  cprints("page_addr");
724  cprinti(page_address);
725  cprints(""); // line return
726  #endif
727 
728  #ifdef DUMP_BOOTPAGE_ON
729  for(i = 0; i < 128; i++){
730  cprinti(current_bootpage.data[i]);
731  }
732  cprints("");
733  cprints("");
734  #endif
735 
736  // Before sending the program, the size of the program is sent to the Kilobot
737  if(page_count == 1){ // the first bootpage has been received
738  IR_setup_message.type = BOOTPGM_SIZE;
739  IR_setup_message.data[0] = page_total;
740  send_IR_message(&IR_setup_message, bootloading_current_cell); // send IR message directly
741  set_all_LEDs(LED_OFF);
742  }
743 
744  // preparing IR messages
745  IR_setup_message.type = BOOTPGM_PAGE;
746  IR_setup_message.data[0] = page_address;
747 
748  /**** Send bootpage to Kilobot ****/
749  for(i = 0; i < BOOTPAGE_SIZE; i += 6){
750  IR_setup_message.data[1] = i / 2;
751  // copy bootload message payload from current_bootpage data
752  for(j = 0; j < 6; j++) {
753  IR_setup_message.data[2 + j] = current_bootpage.data[i + j]; // copy bootpage data in IR message struct to be sent
754  }
755  send_IR_message(&IR_setup_message, bootloading_current_cell); // send IR message directly
756  }
757 
758  sei(); // exit critical section
759 
760  bootpage_sent = 1; // avoids sending the same bootpage twice
761 
762  if((page_address & 0x01) == 0){ // even bootpage number
763  set_LED(bootloading_current_cell, BLUE);
764  }
765  else{
766  set_LED(bootloading_current_cell, GREEN);
767  }
768 
769  if(page_count == page_total){ // last bootpage received
770  page_count = 0; // simply reset counter - will be set to 1 when receiving the first bootpage of the program again
771  set_all_LEDs(LED_OFF);
772  bootloading_current_cell = (bootloading_current_cell + 1) % 4; // jump to the next cell
773  }
774  }
775 
776  break;
777 
778  case MODULE_SETUP:
779  if(!received_setup) {
780  received_setup = 1;
781  setup();
782  }
783 
784  if(module_autostart) {
785  module_state = MODULE_RUNNING;
786  }
787 
788  break;
789 
790  case MODULE_RUNNING:
791  if(!has_started){
792  has_started = 1;
793  // first, broadcast RUN message to the Kilobots for a little while (we assume that all robots are in reset mode already)
794  IR_setup_message.type = RUN;
795  set_all_LEDs(MAGENTA);
796 
797  for(i = 0; i < 10; i++){
798  send_IR_message(&IR_setup_message, CELL_00);
799  send_IR_message(&IR_setup_message, CELL_01);
800  send_IR_message(&IR_setup_message, CELL_02);
801  send_IR_message(&IR_setup_message, CELL_03);
802  }
803 
804  RB_init(CAN_message_tx_buffer); // throw away any previous messages to be sent
805  }
806 
807  /**** EXECUTE USER PROGRAM ****/
808  loop(); // execute main loop defined in the main
809  break;
810  default:
811  break;
812  }
813 
814  if(sending_tracking_data) {
815  if(!sent_tracking_header) {
816  sent_tracking_header = 1;
817 
818  poll_response_message.data[0] = CAN_TRACKING_KILOBOT_START;
819  poll_response_message.data[1] = messages_to_send;
820  poll_response_message.header.length = 2;
821 
822  CAN_message_tx(&poll_response_message, CAN_address_to_dispatcher);
823  _delay_ms(1);
824  }
825 
826  if(messages_to_send > 0) {
827  messages_to_send -= 1;
828 
829  send_next_CAN_message();
830  RB_popfront(CAN_message_tx_buffer);
831  }
