55 static inline void change_LED_state(cell_num_t n, color_t c, brightness_t b) {
56  uint8_t rgb_component; // binary that will successively store R, G and B of the c color
57  uint8_t LED_reg_addr; // contains the address of the PWM register to access
58  uint8_t i; // counter
59 
60  // write each color component of c in the LED driver registers
61  for(i = 0; i < 3; i++) {
62 
63  rgb_component = (c & (1 << i)) >> i; // get the i-th RGB component contained in c (either 0 or 1)
64 
65  LED_reg_addr = GET_LED_REG_ADDR(n, i); // get the address of the current rgb component of subcell s
66 
67  // Ensure that we only change the first five bits of the control register of the PCA9635
68  // and that the changed registers correspond to actual LEDs on the board.
69  if(LED_reg_addr <= LED_UPPER_REG_LIMIT && LED_reg_addr >= LED_LOWER_REG_LIMIT){
70  I2C_start((PCA9635_I2C_ADDR << 1) + I2C_WRITE);
71 
72  //Control register - Autoincrement OFF
73  I2C_write(0x00 | LED_reg_addr); // write address of register on the I2C bus
74  I2C_write(b * rgb_component); // write the value of the PWM (brightness) - will turn the LED off if rgb_component == 0
75 
76  I2C_stop();
77  }
78  // else: no change to the LED driver
79 
80  } // end for
81 
82 
83 } // end function

89 void init_module_LED(void) {
90 
91  uint8_t i = 0; // counter
92 
93  I2C_init();
94 
95  I2C_start((PCA9635_I2C_ADDR << 1) + I2C_WRITE);
96 
97  //Autoincrement ALL registers, start at reg 0 - we're initiating
98  I2C_write(0x80);
99 
100  //Reg 0x00 = MODE1 - set to 0x80 - autoincrement enabled (this is readonly?), do not respond to subaddresses or allcall
101  I2C_write(0x80);
102 
103  //Reg 0x01 = MODE2 - set to 0x02 - noninverted logic state, open-drain
104  I2C_write(0x02);
105 
106  //Reg 0x02-0x11 - PWM0-15 - LED brightnesses, start at 0 brightness
107 
108  for(i = 0; i < 16; i++){
109  I2C_write(0x00);
110  }
111 
112  //Reg 0x12 - Group PWM - should not matter
113  I2C_write(0xff);
114 
115  //Reg 0x13 - Group Freq - should not matter
116  I2C_write(0x00);
117 
118  //Reg 0x14-0x17 - LED Output State Control - all 0xAA (output state for each LED = 0x2)
119  I2C_write(0xaa);
120  I2C_write(0xaa);
121  I2C_write(0xaa);
122  I2C_write(0xaa);
123 
124  //Reg 0x18-0x1b - Subaddressing stuff, doesn't matter, just stop the transfer
125  I2C_stop();
126 
127 }
