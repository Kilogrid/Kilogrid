43 /*
44  * @brief Structure that defines the format of a serial packet.
45  * Packet format: packet types PACKET_STOP, PACKET_LEDTOGGLE, PACKET_FORWARDMSG, PACKET_FORWARDRAWMSG
46  *
47  * PACKET_HEADER
48  * serial_packet_t
49 
50  * data:
51  * [0] IR data payload, byte 0
52  * [1] IR data payload, byte 1
53  * [2] IR data payload, byte 2
54  * [3] IR data payload, byte 3
55  * [4] IR data payload, byte 4
56  * [5] IR data payload, byte 5
57  * [6] IR data payload, byte 6
58  * [7] IR data payload, byte 7
59  * [8] IR data payload, byte 8
60  * [9] IR_message_t
61  * ... 0
62  * [BOOTPAGE_SIZE] 0
63  *
64  * Checksum
65  *
66  * ********
67  * Packet format: packet type PACKET_BOOTPAGE
68  *
69  * PACKET_HEADER
70  * serial_packet_t: PACKET_BOOTPAGE
71  *
72  * data:
73  * [0] Page address (number)
74  * [1] Boot page, byte 0
75  * ...
76  * [BOOTPAGE_SIZE] Boot page, byte 127 (or BOOTPAGE_SIZE - 1)
77  *
78  * Checksum
79  *
80  * @see IR_message_t, serial_packet_type_t
81  */
82 typedef struct __attribute__((__packed__)) {
83  uint8_t header;
84  serial_packet_type_t type;
85  uint8_t data[BOOTPAGE_SIZE + 1]; // additional byte: bootpage number
86  uint8_t checksum;
87 } serial_packet_raw_t;
