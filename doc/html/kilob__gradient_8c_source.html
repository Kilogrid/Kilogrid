109 void kilob_gradient_tracking() {
110  uint8_t action_needed = 0;
111  uint8_t all_escaped = 0;
112  uint8_t i;
113 
114  // If the time is not up, we return
115  if(kilo_ticks < check_timer) {
116  return;
117  }
118 
119  #ifdef TRACKING_LED_DEBUG
120  set_color(RGB(0,0,0));
121  #endif
122 
123  /*
124  * If the last state was turning, we update the minimum and maximum
125  * values, change the state and set up the timer, then return.
126  */
127  if(state == GRADIENT_TRACKING_TURN) {
128  kilob_gradient_update_values();
129  move_straight();
130  state = GRADIENT_TRACKING_TRACK;
131  check_timer = kilo_ticks + check_period;
132  return;
133  }
134 
135  /*
136  * Here we are checking if the minimum and maximum values for a gradient
137  * is the same. That means, that we are moving perpendicular to the
138  * gradient.
139  */
140  for(i=0; i<config_items; i++) {
141  if(!config[i].has_data) {
142  #ifdef TRACKING_LED_DEBUG
143  set_color(RGB(1,1,0));
144  #endif
145  continue;
146  }
147 
148  /*
149  * If the values are different than 0 but the same since we last checked
150  * we indicate that action is needed.
151  */
152  if( config[i].past_minimum != 0 &&
153  config[i].past_maximum != 0 &&
154  config[i].past_minimum == config[i].current_minimum &&
155  config[i].past_maximum == config[i].current_maximum)
156  {
157  action_needed = 1;
158 
159  #ifdef TRACKING_LED_DEBUG
160  set_color(RGB(0,0,1));
161  #endif
162  }
163 
164  /*
165  * If the maximum value is 0, it means that we escaped that gradient.
166  * We reset the detected_at timestamp and call the callback associated
167  * with this gradient.
168  */
169  if(config[i].current_maximum == 0) {
170  config[i].escaped = 1;
171  config[i].detected_at = 0;
172  config[i].on_escape();
173  }
174 
175  /*
176  * Stuck behavior is implemented here. If a gradient can be stuck we
177  * first check if it was detected (value > 0) and then check
178  * for the stuck timeout.
179  *
180  * If we change the turning direction, we update the timer too.
181  */
182  if(config[i].can_stuck == 1 &&
183  config[i].detected_at > 0 &&
184  kilo_ticks - config[i].detected_at > stuck_timeout)
185  {
186  turning_dir = (turning_dir == LEFT) ? RIGHT : LEFT;
187  config[i].detected_at = kilo_ticks;
188  }
189 
190  config[i].has_data = 0;
191  }
192 
193  // Count the number or gradients escaped.
194  for(i=0; i<config_items; i++) {
195  if(config[i].escaped) all_escaped++;
196  }
197 
198  /*
199  * If all gradients were escaped, the next gradient encounter will result in a
200  * random turning direction. This is to prevent choosing the turning direction
201  * randomly all the time. That behavior could result in traversing the
202  * gradient in the wrong direction.
203  */
204  if(all_escaped == config_items) {
205  turning_dir = UNDEFINED;
206 
207  #ifdef TRACKING_LED_DEBUG
208  set_color(RGB(1,1,1));
209  #endif
210  }
211 
212  // Update minimum and maximum values, reset current values
213  kilob_gradient_update_values();
214 
215  // If action is needed we turn a small amount
216  if(action_needed) {
217  kilob_gradient_evade(turn_small_mu, turn_small_sigma);
218  }
219  else {
220  check_timer = kilo_ticks + check_period;
221  }
222 }
