60 
61  CAN_msg_rx = *m; // copy message into CAN_msg_rx buffer struct
62  CAN_msg_type = CAN_msg_rx.data[0];
63 
64  CAN_msg_counter++;
65 
66  current_CAN_msg_number = CAN_msg_rx.data[1];
67 
68  switch(CAN_msg_type){
69 
70  case CAN_CELL_BOOT:
71  asm volatile ("jmp 0x7000"); // jump to bootloader code
72  break;
73 
74  case CAN_CELL_BOOTPAGE:
75  // current_CAN_msg_number is the bootpage data offset, in bytes
76  page_offset = (current_CAN_msg_number * 6) >> 1; // offset of the current data according to the bootpage, in words
77 
78  // Fill in page buffer
79  // The buffer contains words (uint16_t)
80  if(current_CAN_msg_number < (CAN_MESSAGES_PER_BOOTPAGE-1)){ // if the message is not the last CAN message of the bootpage
81  memcpy(page_buffer + page_offset, &CAN_msg_rx.data[2], CAN_MESSAGE_PAYLOAD);
82  page_byte_count += CAN_MESSAGE_PAYLOAD;
83  }
84  else if(current_CAN_msg_number == (CAN_MESSAGES_PER_BOOTPAGE-1)){ // if the message is the last CAN message of current bootpage: only copy the two remaining words (128 - 124 = 4 bytes = 2 words)
85  memcpy(page_buffer + page_offset, &CAN_msg_rx.data[2], CAN_LAST_BOOTPAGE_MESSAGE_REMAINING_BYTES);
86  page_byte_count += CAN_LAST_BOOTPAGE_MESSAGE_REMAINING_BYTES;
87  }
88  //else: ignore message (considered as erroneous)
89 
90  if( page_byte_count >= SPM_PAGESIZE && (!BF_get(page_table, page_address)) ){ // bootpage received and bootpage not already in memory
91 
92  page_byte_count = 0; // reset byte counter
93 
94  // Write program page to flash
95  // Taken from http://www.nongnu.org/avr-libc/user-manual/group__avr__boot.html
96 
97  eeprom_busy_wait();
98 
99  boot_page_erase(page_address*SPM_PAGESIZE);
100  boot_spm_busy_wait();

103  boot_page_fill(page_address*SPM_PAGESIZE+i, page_buffer[j]);
104  }

107  boot_spm_busy_wait();

109  boot_rww_enable();
110 
111  BF_set(page_table, page_address); // set flag to 1 in bitfield to indicate the presence of the bootpage in memory
112 
113  page_count++;
114 
115  if(page_count == page_total){
116  fBootload_finished = 1;
117  }
118 
119  }
120 
121  break;
122 
123  case CAN_CELL_BOOTPAGE_SIZE:
124

126 
127  if (page_count == page_total){
128  fBootload_finished = 1;
129  }
130 
131  break;
132 
133  case CAN_CELL_BOOTPAGE_NUMBER:
134  page_address = CAN_msg_rx.data[2]; // store the address of the bootpage in memory
135  break;
136 
137  case CAN_CELL_IDLE:
138  // return to the application code
139  if(eeprom_read_byte(EEPROM_VALID_APPLICATION_BIT_ADDR) == 1){
140  fBootload_finished = 1;
141  }

143

145  case CAN_CELL_SETUP:
146  // return to the application code
