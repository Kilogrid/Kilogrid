242 static inline void process_message() {
243  AddressPointer_t reset = (AddressPointer_t)0x0000, bootload = (AddressPointer_t)0x7000;
244  calibmsg_t *calibmsg = (calibmsg_t*)&rx_msg.data;
245  if (rx_msg.type < BOOT) {
246  kilo_message_rx(&rx_msg, &rx_dist);
247  return;
248  }
249 
250  if (rx_msg.type != READUID && rx_msg.type != RUN && rx_msg.type != CALIB) {
251  motors_off();
252  }
253 
254  switch (rx_msg.type) {
255  case BOOT:
256  tx_timer_off();
257  bootload();
258  break;
259  case RESET:
260  reset();
261  break;
262  case SLEEP:
263  kilo_state = SLEEPING;
264  break;
265  case WAKEUP:
266  kilo_state = IDLE;
267  break;
268  case CHARGE:
269  kilo_state = CHARGING;
270  break;
271  case VOLTAGE:
272  kilo_state = BATTERY;
273  break;
274  case RUN:
275  if (kilo_state != SETUP && kilo_state != RUNNING) {
276  motors_on();
277  kilo_state = SETUP;
278  }
279  break;
280  case CALIB:
281  switch(calibmsg->mode) {
282  case CALIB_SAVE:
283  if (kilo_state == MOVING) {
284  eeprom_write_byte(EEPROM_UID, kilo_uid&0xFF);
285  eeprom_write_byte(EEPROM_UID+1, (kilo_uid>>8)&0xFF);
286  eeprom_write_byte(EEPROM_LEFT_ROTATE, kilo_turn_left);
287  eeprom_write_byte(EEPROM_RIGHT_ROTATE, kilo_turn_right);
288  eeprom_write_byte(EEPROM_LEFT_STRAIGHT, kilo_straight_left);
289  eeprom_write_byte(EEPROM_RIGHT_STRAIGHT, kilo_straight_right);
290  motors_off();
291  kilo_state = IDLE;
292  }
293  break;
294  case CALIB_UID:
295  kilo_uid = calibmsg->uid;
296  cur_motion = MOVE_STOP;
297  break;
298  case CALIB_TURN_LEFT:
299  if (cur_motion != MOVE_LEFT || kilo_turn_left != calibmsg->turn_left) {
300  prev_motion = MOVE_STOP;
301  cur_motion = MOVE_LEFT;
302  kilo_turn_left = calibmsg->turn_left;
303  }
304  break;
305  case CALIB_TURN_RIGHT:
306  if (cur_motion != MOVE_RIGHT || kilo_turn_right != calibmsg->turn_right) {
307  prev_motion = MOVE_STOP;
308  cur_motion = MOVE_RIGHT;
309  kilo_turn_right = calibmsg->turn_right;
310  }
311  break;
312  case CALIB_STRAIGHT:
313  if (cur_motion != MOVE_STRAIGHT || kilo_straight_right != calibmsg->straight_right || kilo_straight_left != calibmsg->straight_left) {
314  prev_motion = MOVE_STOP;
315  cur_motion = MOVE_STRAIGHT;
316  kilo_straight_left = calibmsg->straight_left;
317  kilo_straight_right = calibmsg->straight_right;
318  }
319  break;
320  }
321  if (calibmsg->mode != CALIB_SAVE && kilo_state != MOVING) {
322  motors_on();
323  kilo_state = MOVING;
324  }
325  break;
326  case READUID:
327  if (kilo_state != MOVING) {
328  motors_on();
329  set_color(RGB(0,0,0));
330  prev_motion = cur_motion = MOVE_STOP;
331  kilo_state = MOVING;
332  }
333 
334  if (kilo_uid&(1<<rx_msg.data[0]))
335  cur_motion = MOVE_LEFT;
336  else
337  cur_motion = MOVE_STOP;
338  break;
339  default:
340  break;
341  }
